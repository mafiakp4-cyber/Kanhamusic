# 🎵 Start command
@app.on_message(filters.command("start") & filters.private)
async def start_cmd(client, message: Message):

        "बस मुझे गाने का नाम भेजो 🎶\n\n"

    )

# 🎵 Text = Song Name
@app.on_message(filters.text & filters.private)
async def song_downloader(client, message: Message):
    query = message.text.strip()
    if not query:
        return

    status = await message.reply_text(f"🔎 Searching for **{query}** ...")

    try:
        # Generate unique filename per user to avoid conflicts
        timestamp = int(time.time())
        filename = f"{message.from_user.id}_{timestamp}.%(ext)s"

        ydl_opts = {
            "format": "bestaudio/best",
            "outtmpl": filename,
            "noplaylist": True,
            "quiet": True,
            # Use browser cookies exported from YouTube
            "cookiefile": "cookies.txt",  # Cookies file path
        }

        # Download in separate thread to avoid blocking
        def download():
            with yt_dlp.YoutubeDL(ydl_opts) as ydl:
                info = ydl.extract_info(f"ytsearch1:{query}", download=True)
                return ydl.prepare_filename(info['entries'][0]), info['entries'][0]['title']

        loop = threading.Thread(target=download)
        loop.start()
        loop.join()

        file_path, title = download()  # Get downloaded file and title

        await message.reply_audio(audio=file_path, title=title, caption=f"🎶 {title}")

        os.remove(file_path)
        await status.delete()

    except Exception as e:
        await status.edit_text(f"❌ Error: {str(e)}")

if __name__ == "__main__":
    threading.Thread(target=run_flask).start()
